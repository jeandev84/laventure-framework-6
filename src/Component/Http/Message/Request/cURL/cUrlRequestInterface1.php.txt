<?php
namespace Laventure\Component\Http\Message\Request\cURL;

interface cUrlRequestInterface
{
    /**
     * @param string $url
     *
     * @param string|null $method
     *
     * @param array $options
     *
     * @return mixed
    */
    public function exec(string $url, string $method = null, array $options = []);
}
